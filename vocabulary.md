<table style="width:100%">
  <tr>

  <tr>
    <td>Tu es</td>
    <td>You are</td> 
  </tr>
   <tr>
    <td>Il/elle/on est</td>
    <td>He/she/it is</td> 
  </tr>
  <tr>
    <td>Nous sommes</td>
    <td>We are</td> 
  </tr>
  <tr>
    <td>Vous etes</td>
